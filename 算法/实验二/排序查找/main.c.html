/* *20140031 陈小龙 计科2班 */ #include <stdio.h> #include <stdlib.h> int swap(int *a,int *b){ int c; c=*a; *a=*b; *b=c; } //这里进行快速排序 int *Partition(int *array,int l,int r){ if(l>=r) return array; int p=array[l],i=l,j=r; while(i<j){ while(i<j && array[j]>=p) j--; swap(&array[i],&array[j]); while(i<j && array[i]<=p) i++; swap(&array[j],&array[i]); } Partition(array,l,i-1); Partition(array,i+1,r); return array; } //这里进行折半查找 int ersearch(int array[],int n,int k){ int low=0,high=n-1,m; while(low<=high){ m=(low+high)/2; if(k==array[m]) return m; else if(k>array[m]) low=m+1; else high=m-1; } return -1; } int main() { int i=0,a[]={56,87,789,54,12,35,11,102,456,7821,2,44,245,56,879,45,82,57,62,30},k; int *b; b=Partition(a,0,19); printf("快速排序后的数组为:"); for(i=0;i<20;i++){ printf("%d ",b[i]); } printf("\n输入要查找的数:"); scanf("%d",&k); printf("该数对应的下标值是：%d\n",ersearch(a,20,k)); return 0; }
